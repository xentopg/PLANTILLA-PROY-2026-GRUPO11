#include <Adafruit_Fingerprint.h>
#include <Servo.h>
#include <Wire.h>
#include <Adafruit_PN532.h>

// Usamos Serial1 para el sensor en la Arduino UNO Q (Pines RX1 y TX1)
Adafruit_Fingerprint finger = Adafruit_Fingerprint(&Serial1);
Servo myservo;

// Inicializamos el sensor NFC en los pines SDA (D20) y SCL (D21)
Adafruit_PN532 nfc(20, 21);

const int SERVO_PIN = 9; // Pin digital para la señal del servo
const int LED_PIN = 7;   // Pin digital asignado para el LED verde

// Definición de Tarjetas NFC autorizadas y denegadas
const uint8_t TARJETA_A[] = {0xDE, 0x14, 0x29, 0x6E};
const uint8_t TARJETA_B[] = {0x0E, 0xE9, 0x1B, 0x6E};
const uint8_t TARJETA_C[] = {0x1C, 0x3F, 0x04, 0x30};
const uint8_t CELU_VICENTE[] = {0x01, 0x02, 0x03, 0x04};
const uint8_t AMIIBO_PEACH[] = {0x04, 0xF3, 0x09, 0x92, 0x99, 0x3C, 0x84};

void setup() {
  Serial.begin(9600);
  
  // Espera fija de seguridad para evitar bloqueos en el Monitor Serie de Arduino Lab
  delay(3000); 
  Serial.println("\n--- XENTLOCK: SISTEMA DE ACCESO CONTROLADO ---");

  // Configurar el pin del LED como salida
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW); // Asegurar que inicie apagado

  // Inicializar el Servomotor
  myservo.attach(SERVO_PIN);
  myservo.write(180); // Estado inicial: Bloqueado por defecto a 180°
  Serial.println("[INFO] Servomotor inicializado y BLOQUEADO (180°). LED apagado.");

  // Configurar la velocidad del sensor de huellas AS608
  Serial1.begin(57600);
  delay(500);
  
  if (finger.verifyPassword()) {
    Serial.println("[OK] Sensor de huellas AS608 detectado.");
  } else {
    Serial.println("[ERROR] No se encuentra el sensor de huellas :(");
    while (1) { delay(1); }
  }

  // --- NUEVO: Inicialización del sensor NFC ---
  nfc.begin();
  uint32_t versiondata = nfc.getFirmwareVersion();
  if (versiondata) {
    Serial.println("[OK] Sensor NFC PN532 detectado.");
    nfc.SAMConfig(); // Configura el módulo para leer etiquetas RFID
  } else {
    Serial.println("[ERROR] No se encuentra el sensor NFC PN532 :(");
  }

  Serial.println("Esperando que se coloque un dedo sobre el cristal...");
}

void loop() {
  // Intentar obtener la lectura del dedo de forma continua
  int result = getFingerprintID();
  
  // Si el resultado es una coincidencia válida (ID mayor que 0)
  if (result > 0) {
    Serial.print("[ACCESO CONCEDIDO] Bienvenido Usuario ID #");
    Serial.println(result);
    
    // ACCIÓN DE APERTURA E INDICACIÓN LUMINOSA
    Serial.println("Desbloqueando cerrojo... Moviendo a 90° y encendiendo LED.");
    digitalWrite(LED_PIN, HIGH); // Enciende el LED verde
    myservo.write(90);           // Mueve el servo a 90°
    
    // Mantener abierto y encendido durante 4 segundos completos (Nota: tu delay dice 7000ms)
    delay(7000); 
    
    // ACCIÓN DE CIERRE Y APAGADO AUTOMÁTICO
    Serial.println("Tiempo agotado. Cerrando cerrojo... Retornando a 180° y apagando LED.");
    myservo.write(180);          // Retorna el servo a 180°
    digitalWrite(LED_PIN, LOW);  // Apaga el LED verde
    Serial.println("\nEsperando nueva huella...");
  }
  
  // --- NUEVO: Revisar si se aproxima una tarjeta NFC en esta vuelta ---
  revisarTarjetaNFC();
  
  delay(50); // Pequeña pausa para no saturar el procesador
}

// Función auxiliar para procesar y validar de forma continua
int getFingerprintID() {
  uint8_t p = finger.getImage();
  if (p != FINGERPRINT_OK) return -1; // No hay dedo o hay error de lectura

  p = finger.image2Tz();
  if (p != FINGERPRINT_OK) return -1; // Error al convertir la imagen

  p = finger.fingerFastSearch();
  if (p == FINGERPRINT_OK) {
    // Retorna el ID de la huella que coincide en la memoria flash
    return finger.fingerID; 
  } else {
    Serial.println("[ACCESO DENEGADO] Huella no registrada en el sistema.");
    return -1;
  }
}

// --- NUEVO: Función auxiliar para controlar el acceso mediante el sensor NFC ---
void revisarTarjetaNFC() {
  uint8_t success;
  uint8_t uid[] = { 0, 0, 0, 0, 0, 0, 0 };  
  uint8_t uidLength;                        

  // Escuchamos el bus con un timeout ultra-corto (30ms) para que no interfiera con la fluidez del lector de huellas
  success = nfc.readPassiveTargetID(PN532_MIFARE_ISO14443A, uid, &uidLength, 30);
  
  if (success) {
    if (uidLength == 4) { // Validamos tarjetas estándar de 4 bytes
      
      // CASO 1: TARJETA A (Apertura exitosa)
      if (memcmp(uid, TARJETA_A, 4) == 0) {
        Serial.println("\n[ACCESO CONCEDIDO] Tarjeta A detectada.");
        ejecutarAperturaNFC();
      } 
      // CASO 2: TARJETA C (Apertura exitosa)
      else if (memcmp(uid, TARJETA_C, 4) == 0) {
        Serial.println("\n[ACCESO CONCEDIDO] Tarjeta C detectada.");
        ejecutarAperturaNFC();
      } 
      // NUEVO CASO: CELU VICENTE PAVEZ (Apertura exitosa)
      else if (memcmp(uid, CELU_VICENTE, 4) == 0) {
        Serial.println("\n[ACCESO CONCEDIDO] Celu Vicente Pavez detectado.");
        ejecutarAperturaNFC();
      }
      // CASO 3: TARJETA B (Acceso Denegado de forma estricta)
      else if (memcmp(uid, TARJETA_B, 4) == 0) {
        Serial.println("\n[ACCESO DENEGADO] Tarjeta B bloqueada. No autorizada.");
        // Alerta visual sin mover el servo
        for(int i = 0; i < 3; i++) {
          digitalWrite(LED_PIN, HIGH); delay(150);
          digitalWrite(LED_PIN, LOW);  delay(150);
        }
      } 
      // CUALQUIER OTRA TARJETA NO DECLARADA
      else {
        Serial.println("\n[ACCESO DENEGADO] Tarjeta NFC desconocida.");
      }
    }
    // NUEVA VALIDACIÓN: Si el sensor lee una tarjeta de 7 bytes (como el Amiibo)
    else if (uidLength == 7) {
      if (memcmp(uid, AMIIBO_PEACH, 7) == 0) {
        Serial.println("\n[ACCESO CONCEDIDO] Amiibo Peach detectado.");
        ejecutarAperturaNFC();
      } else {
        Serial.println("\n[ACCESO DENEGADO] Tag NFC de 7 bytes desconocido.");
      }
    }
  }
}

// --- NUEVO: Subfunción para replicar exactamente tu rutina de apertura ---
void ejecutarAperturaNFC() {
  Serial.println("Desbloqueando cerrojo... Moviendo a 90° y encendiendo LED.");
  digitalWrite(LED_PIN, HIGH); 
  myservo.write(90);           
  
  delay(7000); // Mantiene tus 7 segundos completos de apertura de la huella
  
  Serial.println("Tiempo agotado. Cerrando cerrojo... Retornando a 180° y apagando LED.");
  myservo.write(180);          
  digitalWrite(LED_PIN, LOW);  
  Serial.println("\nEsperando nueva huella o tarjeta...");
}
